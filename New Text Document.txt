                        mainVideo.pause();
                        // If image thumb
                        if (thumb.dataset.type === 'img') {
                            mainImg.src = thumb.dataset.src;
                            mainImg.style.opacity = '1';
                            mainVideo.style.display = 'none';
                            mainImg.style.display = 'block';
                        }
                        // If video thumb
                        if (thumb.dataset.type === 'video') {
                            mainVideo.querySelector('source').src = thumb.dataset.src;
                            mainVideo.load();
                            mainVideo.style.display = 'block';
                            mainImg.style.display = 'none';
                            mainVideo.play().catch(() => {});
                        }
                    });
                });
            });

            // Intersection Observer: only play videos in view for smoothness
            if ('IntersectionObserver' in window) {
                const observer = new IntersectionObserver(entries => {
                    entries.forEach(entry => {
                        const video = entry.target;
                        if (entry.isIntersecting) {
                            video.play().catch(() => {});
                        } else {
                            video.pause();
                        }
                    });
                }, { threshold: 0.5 });

                document.querySelectorAll('#employees .service-video').forEach(video => {
                    observer.observe(video);
                });
            }

            function setModalVideoSources(base) {
                const sources = modalVideo.querySelectorAll('source');
                if (sources.length >= 3 && base) {
                    sources[0].src = base.replace(/\.\w+$/, '.mp4');
                    sources[1].src = base.replace(/\.\w+$/, '.webm');
                    sources[2].src = base.replace(/\.\w+$/, '.ogv');
                } else {
                    sources.forEach(s => s.src = '');
                }
                modalVideo.load();
            }

            videoCards.forEach((card, idx) => {
                card.addEventListener('click', () => {
                    pauseAllInline();
                    const src = card.getAttribute('data-video');
                    setModalVideoSources(src);
                    modalVideo.currentTime = 0;
                    modalVideo.muted = false;
                    modalVideo.play().catch(()=>{});
                    // Set label
                    const label = card.querySelector('div') ? card.querySelector('div').textContent : '';
                    modalLabel.textContent = label;
                    videoModal.classList.add('active');
                    document.body.style.overflow = 'hidden';

        // Add focus management for accessibility
        const focusableElements = 'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])';

        function trapFocus(element) {
            const focusableContent = element.querySelectorAll(focusableElements);
            const firstFocusableElement = focusableContent[0];
            const lastFocusableElement = focusableContent[focusableContent.length - 1];

            element.addEventListener('keydown', function(e) {
                if (e.key === 'Tab') {
                    if (e.shiftKey) {
                        if (document.activeElement === firstFocusableElement) {
                            lastFocusableElement.focus();
                            e.preventDefault();
                        }
                    } else {
                        if (document.activeElement === lastFocusableElement) {
                            firstFocusableElement.focus();
                            e.preventDefault();
                        }
                    }
                }
            });
        }

        // Apply focus trap to modal when it opens
        const originalShowModal = showModal;
        showModal = function(content) {
            originalShowModal(content);
            trapFocus(modal);
            // Focus first focusable element in modal
            const firstFocusable = modal.querySelector(focusableElements);
            if (firstFocusable) {
                setTimeout(() => firstFocusable.focus(), 100);
            }
        };

        // Enhanced form validation
        function validateForm() {
            const form = document.getElementById('bookingForm');
            const inputs = form.querySelectorAll('input[required], select[required]');
            let isValid = true;

            inputs.forEach(input => {
                if (!input.value.trim()) {
                    input.classList.add('error');
                    isValid = false;
                } else {
                    input.classList.remove('error');
                }
            });

            // Email validation
            const emailInput = document.getElementById('email');
            if (emailInput.value && !isValidEmail(emailInput.value)) {
                emailInput.classList.add('error');
                isValid = false;
            }

            // Phone validation
            const phoneInput = document.getElementById('phone');
            if (phoneInput.value && !isValidPhone(phoneInput.value)) {
                phoneInput.classList.add('error');
                isValid = false;
            }

            return isValid;
        }

        function isValidEmail(email) {
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            return emailRegex.test(email);
        }

        function isValidPhone(phone) {
            const phoneRegex = /^[\+]?[0-9\s\-\(\)]{10,}$/;
            return phoneRegex.test(phone);
        }
